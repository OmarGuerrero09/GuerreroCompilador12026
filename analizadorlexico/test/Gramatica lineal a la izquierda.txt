<Programa> -> <Bloque> .

<Bloque> -> <LinealC> <LinealV> <LinealP> <Proposicion>

<LinealC> -> <LinealConst>
<LinealC> -> ∅
<LinealConst> -> const <CicloNum> ;

<CicloNum> -> <CicloNum> , id = num
<CicloNum> -> id = num

<LinealV> -> <LinealVar>
<LinealV> -> ∅
<LinealVar> -> var <CicloId> ;

<CicloId> -> <CicloId> , id
<CicloId> -> id

<LinealP> -> <CicloProc>
<LinealP> -> ∅
<CicloProc> -> <CicloProc> <LinealProc>
<CicloProc> -> <LinealProc>
<LinealProc> -> proced id ; <Bloque> ;

<Condicion> -> <Expresion> <MultSigno> <Expresion>

<MultSigno> -> ==
<MultSigno> -> !=
<MultSigno> -> <
<MultSigno> -> >
<MultSigno> -> <=
<MultSigno> -> >=

<Proposicion> -> <MultPropo>

<MultPropo> -> <LinealBegin>
<MultPropo> -> <LinealId>
<MultPropo> -> <LinealWrite>
<MultPropo> -> <LinealRead>
<MultPropo> -> <LinealCall>
<MultPropo> -> <LinealIf>
<MultPropo> -> <LinealWhile>
<MultPropo> -> <LinealFor>

<LinealBegin> -> begin <CicloProp> end

<CicloProp> -> <CicloProp> ; <Proposicion>
<CicloProp> -> <Proposicion>

<LinealId> -> id = <Expresion>
<LinealWrite> -> write <DisyWri>

<DisyWri> -> id
<DisyWri> -> num

<LinealRead> -> read id
<LinealCall> -> call id
<LinealIf> -> if <Condicion> then <Proposicion>
<LinealWhile> -> while <Condicion> do <Proposicion>
<LinealFor> -> for id = <Expresion> <DisyExp> <Expresion> do <Proposicion>

<DisyExp> -> to
<DisyExp> -> down

<Expresion> -> <Expresion> <CicloTerm> <Termino>
<Expresion> -> <Termino>

<CicloTerm> -> <DisySig>
<DisySig> -> +
<DisySig> -> -

<Termino> -> <Termino> <CicloFact> <Factor>
<Termino> -> <Factor>

<CicloFact> -> <DisySigno>
<DisySigno> -> *
<DisySigno> -> /

<Factor> -> <MultExp>

<MultExp> -> (<Expresion>)
<MultExp> -> id
<MultExp> -> num
